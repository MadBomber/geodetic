def initialize(name:)
  @name = name.upcase
  raise NameError, "Unknown datum: #{@name}" unless ELLIPSES.key?(@name)
  data   = ELLIPSES[@name]
  @desc  = data[:desc]
  @a     = data[:a]
  @f_inv = data[:f_inv]
  @f     = 1.0 / @f_inv
  @b     = data[:b]
  @e2    = data[:e2]
  @e     = Math.sqrt(@e2)
end

def self.get(datum_name)
  name = datum_name.upcase
  raise NameError, "Unknown datum: #{name}" unless ELLIPSES.key?(name)
  data = ELLIPSES[name]
  {
    'name'  => name,
    'desc'  => data[:desc],
    'a'     => data[:a],
    'f_inv' => data[:f_inv],
    'f'     => 1.0 / data[:f_inv],
    'b'     => data[:b],
    'e2'    => data[:e2],
    'e'     => Math.sqrt(data[:e2])
  }
end
